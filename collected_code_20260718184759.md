## src/components/Hero/Hero.tsx

```plaintext
"use client";

import { useRef, useState, useEffect } from "react";
import gsap from "gsap";
import { ScrollTrigger } from "gsap/ScrollTrigger";
import { useGSAP } from "@gsap/react";
import styles from "./Hero.module.css";

// Регистрируем плагин строго один раз
if (typeof window !== "undefined") {
  gsap.registerPlugin(ScrollTrigger);
}

export default function Hero() {
  const containerRef = useRef<HTMLElement>(null);
  const textRef = useRef<HTMLDivElement>(null);
  const pathRef = useRef<SVGPathElement>(null);
  const [width, setWidth] = useState<number>(0);

  // O(1) подписка на ресайз для точной математики Безье
  useEffect(() => {
    const handleResize = () => setWidth(window.innerWidth);
    handleResize(); // Инициализация
    window.addEventListener("resize", handleResize);
    return () => window.removeEventListener("resize", handleResize);
  }, []);

  useGSAP(() => {
    if (width === 0 || !pathRef.current || !containerRef.current || !textRef.current) return;

    // Математика точек пути
    // Начало и конец (ровная линия в нулевой координате Y)
    const flatPath = `M 0 0 L ${width} 0 Q ${width / 2} 0 0 0 Z`;
    // Пик натяжения (опускается на 300px вниз по центру)
    const curvePath = `M 0 0 L ${width} 0 Q ${width / 2} 300 0 0 Z`;

    // Устанавливаем изначальное ровное состояние
    gsap.set(pathRef.current, { attr: { d: flatPath } });

    // Создаем таймлайн, жестко привязанный к скроллу секции
    const tl = gsap.timeline({
      scrollTrigger: {
        trigger: containerRef.current,
        start: "top top", // Начинаем, когда секция на самом верху экрана
        end: "bottom top", // Заканчиваем, когда низ секции дойдет до верха экрана
        scrub: true, // Плавная интерполяция вместе с Lenis
      },
    });

    // Фаза 1: Параллакс текста (уезжает вниз медленнее, чем фон)
    tl.to(textRef.current, {
      y: 250,
      ease: "none",
    }, 0);

    // Фаза 2: Натягивание и отпускание кривой Безье
    // Используем относительное время (0 до 0.5 — натяжение, 0.5 до 1 — возврат)
    tl.to(pathRef.current, {
      attr: { d: curvePath },
      ease: "power2.in",
    }, 0)
    .to(pathRef.current, {
      attr: { d: flatPath },
      ease: "power2.out",
    }, ">"); // ">" означает старт сразу после предыдущей анимации

  }, { dependencies: [width], scope: containerRef });

  return (
    <section ref={containerRef} className={styles.hero}>
      <div ref={textRef} className={styles.content}>
        <h1 className={styles.title}>
          Studia 74<span className={styles.dot}>.</span>
        </h1>
      </div>

      {/* SVG контейнер прикреплен к нижней границе секции */}
      {width > 0 && (
        <svg className={styles.svgContainer}>
          <path ref={pathRef} className={styles.svgPath} />
        </svg>
      )}
    </section>
  );
}
```
